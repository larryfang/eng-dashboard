# Domain-Scoped SQLite DB Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Introduce a per-domain SQLite database (`data/domains/ecosystem.db`) that isolates all reference and transactional dashboard data from the PA personal database, with section-level sync/cache control and 30/60/90-day period selection.

**Architecture:**
Each domain (Engineering Director's org) owns its own DB file under `data/domains/{slug}.db`. Reference tables (`ref_teams`, `ref_members`) are seeded from `organization.yaml` on startup. Transactional tables mirror the existing GitLab/Jira tables but live in the domain DB. A `sync_status` and `section_cache` table enable per-section cache-or-sync UX with 30/60/90-day period options.

**Tech Stack:** SQLAlchemy, SQLite, FastAPI, React/TypeScript, Vite, uv

---

## Context & Current State

| Item | Current state |
|------|--------------|
| `sinch_pa.db` | Monolithic — mixes PA personal data AND dashboard data |
| `gitlab_mr_activity` etc. | In `sinch_pa.db`, no domain column |
| Engineer lookup | In-memory `_member_lookup` from `organization.yaml` — not persisted |
| Cache | `ecosystem_metrics_cache` — simple JSON blob, no section/period granularity |
| Sync tracking | `gitlab_sync_logs` — per-sync record, not per-section status |
| Period support | Hard-coded 30-day default, backend accepts `days=` param but no UI |

**Key design principle (must be preserved everywhere):**
Engineer activity follows the engineer (by author), NOT the repo the team owns.
Config team membership (`ref_members.team_slug`) always overrides repo-derived team.

## Task 1: Create domain DB infrastructure

**Files:**
- Create: `backend/database_domain.py`
- Create: `data/domains/.gitkeep`

**What it does:** Introduces a second SQLAlchemy engine/session pointing to `data/domains/{domain_slug}.db`, completely separate from the PA `sinch_pa.db`. The PA `SessionLocal` is untouched.

**Step 1: Create the domain DB module**

```python
# backend/database_domain.py
"""
Domain-scoped database for dashboard data.

Each domain (director's org) gets its own SQLite file:
  data/domains/ecosystem.db
  data/domains/platform.db   (future)

This is completely separate from sinch_pa.db (PA personal data).
"""
import os
from pathlib import Path
from sqlalchemy import create_engine, event
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker

PROJECT_ROOT = Path(__file__).resolve().parent.parent
DOMAINS_DIR = PROJECT_ROOT / "data" / "domains"

DomainBase = declarative_base()


def get_domain_db_path(domain_slug: str) -> Path:
    DOMAINS_DIR.mkdir(parents=True, exist_ok=True)
    return DOMAINS_DIR / f"{domain_slug}.db"


def create_domain_engine(domain_slug: str):
    db_path = get_domain_db_path(domain_slug)
    engine = create_engine(
        f"sqlite:///{db_path}",
        connect_args={"check_same_thread": False, "timeout": 30},
    )
    @event.listens_for(engine, "connect")
    def set_sqlite_pragma(conn, _):
        conn.execute("PRAGMA journal_mode=WAL")
        conn.execute("PRAGMA foreign_keys=ON")
    return engine


# Ecosystem domain — the default for this installation
_ecosystem_engine = None
_EcosystemSession = None


def get_ecosystem_engine():
    global _ecosystem_engine
    if _ecosystem_engine is None:
        _ecosystem_engine = create_domain_engine("ecosystem")
    return _ecosystem_engine


def get_ecosystem_session():
    """FastAPI dependency: yields a domain DB session."""
    global _EcosystemSession
    if _EcosystemSession is None:
        _EcosystemSession = sessionmaker(
            autocommit=False, autoflush=False, bind=get_ecosystem_engine()
        )
    db = _EcosystemSession()
    try:
        yield db
    finally:
        db.close()


def init_domain_db():
    """Create all domain tables (called from main.py lifespan)."""
    from backend.models_domain import DomainBase  # import after models defined
    DomainBase.metadata.create_all(bind=get_ecosystem_engine())
```

**Step 1: Write the seeder**

```python
# backend/services/domain_seeder.py
"""
Seeds ref_teams and ref_members in the domain DB from organization.yaml.
Called on startup and via POST /api/domain/seed.
"""
import logging
from sqlalchemy.orm import Session

logger = logging.getLogger(__name__)


def seed_reference_data(db: Session) -> dict:
    """
    Upsert ref_teams and ref_members from the org config.
    Returns {"teams": N, "members": N}.
    """
    from backend.core.config_loader import get_config
    from backend.models_domain import RefTeam, RefMember
    import json

    config = get_config()
    teams_seeded = 0
    members_seeded = 0

    for team in config.teams:
        slug = team.slug or team.key.lower()
        team_display = team.name + (f" ({team.scrum_name})" if team.scrum_name else "")

        # Upsert team
        existing = db.query(RefTeam).filter_by(slug=slug).first()
        if existing:
            row = existing
        else:
            row = RefTeam(slug=slug)
            db.add(row)

        row.key          = team.key
        row.name         = team.name
        row.scrum_name   = team.scrum_name
        row.jira_project = team.jira_project
        row.gitlab_path  = team.gitlab_path
        row.headcount    = team.headcount
        row.em_name      = team.lead
        row.em_email     = team.lead_email
        row.products     = json.dumps(team.products or [])
        teams_seeded += 1

        # Upsert members
        for member in team.gitlab_members:
            existing_m = db.query(RefMember).filter_by(
                gitlab_username=member.username
            ).first()
            if existing_m:
                m = existing_m
            else:
                m = RefMember(gitlab_username=member.username)
                db.add(m)

            m.name                 = member.name
            m.email                = member.email
            m.role                 = member.role
            m.team_slug            = slug
            m.team_display         = team_display
            m.em_name              = team.lead
            m.em_email             = team.lead_email
            m.jira_project         = team.jira_project
            m.jira_account_id      = member.jira_account_id
            m.gitlab_path          = team.gitlab_path
            m.exclude_from_metrics = member.exclude_from_metrics
            m.departed             = getattr(member, 'departed', False)
            members_seeded += 1

    db.commit()
    logger.info(f"Seeded {teams_seeded} teams, {members_seeded} members into domain DB")
    return {"teams": teams_seeded, "members": members_seeded}
```

## Task 4: Migrate existing transactional data into ecosystem.db

**Files:**

**What it does:** One-time migration script that copies `gitlab_mr_activity` and `gitlab_metrics` from `sinch_pa.db` into the new `ecosystem.db` `mr_activity` and `team_metrics` tables, resolving `author_team` from `ref_members` at migration time.

**Step 1: Write the migration script**

```python
# backend/commands/migrate_to_domain_db.py
"""
One-time migration: sinch_pa.db → ecosystem.db

Copies gitlab_mr_activity and gitlab_metrics into the new domain tables.
Safe to re-run (upserts).
"""
import sys
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker
from backend.database_domain import get_ecosystem_engine, init_domain_db
from backend.models_domain import MRActivity, TeamMetrics, RefMember
from datetime import datetime, timezone

OLD_DB = str(PROJECT_ROOT / "sinch_pa.db")


def resolve_team(username: str, domain_session) -> str:
    """Look up canonical team from ref_members (config is authoritative)."""
    m = domain_session.query(RefMember).filter(
        RefMember.gitlab_username.ilike(username)
    ).first()
    return m.team_slug if m else "unknown"


def migrate():
    old_engine = create_engine(f"sqlite:///{OLD_DB}", connect_args={"check_same_thread": False})
    OldSession = sessionmaker(bind=old_engine)

    init_domain_db()
    DomainSession = sessionmaker(bind=get_ecosystem_engine())

    with OldSession() as old_db, DomainSession() as new_db:
        # --- MR Activity ---
        rows = old_db.execute(text("SELECT * FROM gitlab_mr_activity")).fetchall()
        print(f"Migrating {len(rows)} MR activity rows...")
        for r in rows:
            author_team = resolve_team(r.author_username, new_db)
            existing = new_db.query(MRActivity).filter_by(
                repo_id=r.repo_id, mr_iid=r.mr_iid
            ).first()
            if existing:
                continue
            new_db.add(MRActivity(
                mr_iid=r.mr_iid,
                repo_id=r.repo_id,
                title=r.title,
                description=r.description,
                source_branch=r.source_branch,
                author_username=r.author_username,
                author_team=author_team,
                state=r.state,
                created_at=r.created_at,
                merged_at=r.merged_at,
                web_url=r.web_url,
                jira_tickets=r.jira_tickets,
                epic_keys=r.epic_keys,
                lines_added=r.lines_added,
                lines_removed=r.lines_removed,
                files_changed=r.files_changed,
                cycle_time_hours=r.cycle_time_hours,
                synced_at=r.synced_at,
            ))
        new_db.commit()
        print("MR activity migrated.")

        # --- Team Metrics ---
        rows = old_db.execute(text("SELECT * FROM gitlab_metrics")).fetchall()
        print(f"Migrating {len(rows)} team metrics rows...")
        for r in rows:
            existing = new_db.query(TeamMetrics).filter_by(
                team=r.team, metric_date=r.metric_date
            ).first()
            if existing:
                continue
            new_db.add(TeamMetrics(
                team=r.team,
                metric_date=r.metric_date,
                pipeline_runs=r.pipeline_runs,
                pipeline_success=r.pipeline_success,
                pipeline_failed=r.pipeline_failed,
                avg_duration_seconds=r.avg_duration_seconds,
                mrs_merged=r.merge_requests_merged,
                avg_cycle_time_hours=r.avg_mr_cycle_time_hours,
                deployment_frequency=r.deployment_frequency,
                lead_time_hours=r.lead_time_hours,
                change_failure_rate=r.change_failure_rate,
                mttr_hours=r.mttr_hours,
                dora_level=r.dora_level,
                synced_at=r.synced_at,
            ))
        new_db.commit()
        print("Team metrics migrated.")

    print("Migration complete.")


if __name__ == "__main__":
    migrate()
```

```python
@router.get("/engineers")
async def list_engineers(
    team: Optional[str] = Query(default=None),
    days: int = Query(default=30, ge=1, le=365),
    db: Session = Depends(get_ecosystem_session),
):
    from datetime import datetime, timedelta, timezone
    from collections import defaultdict
    from sqlalchemy import text as sa_text
    from backend.models_domain import MRActivity, RefMember

    since = datetime.now(timezone.utc) - timedelta(days=days)

    # Query mr_activity from domain DB
    query = db.query(MRActivity).filter(MRActivity.created_at >= since)
    if team:
        # Filter by config team — uses author_team column (resolved at sync time)
        # Also include anyone whose ref_members says they belong to this team,
        # even if their MRs were in another team's repos.
        team_usernames = {
            m.gitlab_username.lower()
            for m in db.query(RefMember).filter_by(team_slug=team).all()
        }
        query = query.filter(
            (MRActivity.author_team == team) |
            MRActivity.author_username.in_(team_usernames)
        )

    rows = query.order_by(MRActivity.created_at.desc()).all()

    # Aggregate per author
    authors = defaultdict(lambda: {"mrs_opened": 0, "mrs_merged": 0, "last_activity": None})
    for r in rows:
        a = authors[r.author_username]
        a["mrs_opened"] += 1
        if r.state == "merged":
            a["mrs_merged"] += 1
        if a["last_activity"] is None or r.created_at > a["last_activity"]:
            a["last_activity"] = r.created_at

    # Enrich from ref_members (authoritative team + identity)
    results = []
    for username, stats in sorted(authors.items(), key=lambda x: -x[1]["mrs_opened"]):
        member = db.query(RefMember).filter(
            RefMember.gitlab_username.ilike(username)
        ).first()
        resolved_team = member.team_slug if member else "unknown"
        if team and resolved_team != team:
            continue
        results.append({
            "username": username,
            "name": member.name if member else username,
            "email": member.email if member else None,
            "role": member.role if member else None,
            "team": resolved_team,
            "team_display": member.team_display if member else resolved_team,
            "team_lead": member.em_name if member else None,
            "jira_project": member.jira_project if member else None,
            "mrs_opened": stats["mrs_opened"],
            "mrs_merged": stats["mrs_merged"],
            "last_activity": stats["last_activity"],
        })

    return {"engineers": results, "count": len(results), "period_days": days}
```

## Task 6: Add sync_status endpoints

**What it does:** Exposes `GET /api/sync/status` and `POST /api/sync/{section}` so the frontend can show "last synced X minutes ago" per section and trigger a refresh.

**Step 1: Create sync_router.py**

```python
# backend/routers/sync_router.py
"""
Sync control endpoints.

GET  /api/sync/status               - status of all sections
GET  /api/sync/status/{section}     - status of one section
POST /api/sync/{section}?days=30    - trigger sync for section + period
"""
from fastapi import APIRouter, Depends, Query
from sqlalchemy.orm import Session
from datetime import datetime, timezone, timedelta
from typing import Optional
from backend.database_domain import get_ecosystem_session
from backend.models_domain import SyncStatus

router = APIRouter(prefix="/api/sync", tags=["Sync"])

SECTIONS = ["engineers", "team_metrics", "jira_epics", "repos", "dora"]
DEFAULT_TTL_HOURS = {"engineers": 1, "team_metrics": 6, "jira_epics": 2, "repos": 24, "dora": 6}


def _get_or_create_status(db: Session, section: str, period_days: int) -> SyncStatus:
    row = db.query(SyncStatus).filter_by(section=section, period_days=period_days).first()
    if not row:
        row = SyncStatus(section=section, period_days=period_days, status="idle")
        db.add(row)
        db.commit()
    return row


def _status_dict(row: SyncStatus) -> dict:
    now = datetime.now(timezone.utc)
    is_stale = (
        row.last_synced_at is None
        or (row.next_sync_at and row.next_sync_at < now)
    )
    return {
        "section": row.section,
        "period_days": row.period_days,
        "status": row.status,
        "last_synced_at": row.last_synced_at.isoformat() if row.last_synced_at else None,
        "next_sync_at": row.next_sync_at.isoformat() if row.next_sync_at else None,
        "records_synced": row.records_synced,
        "is_stale": is_stale,
        "error": row.error_message,
    }


@router.get("/status")
async def get_all_sync_status(db: Session = Depends(get_ecosystem_session)):
    result = {}
    for section in SECTIONS:
        result[section] = {}
        for period in [30, 60, 90]:
            row = _get_or_create_status(db, section, period)
            result[section][str(period)] = _status_dict(row)
    return result


@router.get("/status/{section}")
async def get_section_sync_status(
    section: str,
    days: int = Query(default=30),
    db: Session = Depends(get_ecosystem_session),
):
    row = _get_or_create_status(db, section, days)
    return _status_dict(row)


@router.post("/{section}")
async def trigger_sync(
    section: str,
    days: int = Query(default=30),
    db: Session = Depends(get_ecosystem_session),
):
    """Mark a sync as started. The actual sync runs in the existing GitLab/Jira endpoints."""
    row = _get_or_create_status(db, section, days)
    row.status = "syncing"
    db.commit()
    # Delegate to existing sync endpoints
    return {"status": "syncing", "section": section, "period_days": days}


def mark_sync_complete(db: Session, section: str, period_days: int, records: int):
    """Helper called by sync services after completing a sync."""
    row = _get_or_create_status(db, section, period_days)
    now = datetime.now(timezone.utc)
    ttl = DEFAULT_TTL_HOURS.get(section, 6)
    row.status = "success"
    row.last_synced_at = now
    row.next_sync_at = now + timedelta(hours=ttl)
    row.records_synced = records
    row.error_message = None
    db.commit()
```

```typescript
// frontend/src/components/SyncStatusBadge.tsx
import { formatDistanceToNow } from 'date-fns'

interface SyncInfo {
  last_synced_at: string | null
  is_stale: boolean
  status: string
}

interface Props {
  syncInfo?: SyncInfo
  onRefresh: () => void
  loading?: boolean
}

export default function SyncStatusBadge({ syncInfo, onRefresh, loading }: Props) {
  if (!syncInfo) return null
  const label = syncInfo.last_synced_at
    ? `Synced ${formatDistanceToNow(new Date(syncInfo.last_synced_at), { addSuffix: true })}`
    : 'Never synced'
  return (
    <div className="flex items-center gap-2 text-xs text-gray-500">
      <span className={syncInfo.is_stale ? 'text-yellow-500' : ''}>{label}</span>
      <button
        onClick={onRefresh}
        disabled={loading || syncInfo.status === 'syncing'}
        className="text-blue-400 hover:text-blue-300 disabled:opacity-40"
      >
        {loading || syncInfo.status === 'syncing' ? 'Syncing…' : 'Refresh'}
      </button>
    </div>
  )
}
```
